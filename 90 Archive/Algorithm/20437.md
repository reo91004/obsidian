import sys


def main() -> None:
    input = sys.stdin.readline
    T = int(input().strip())

    for _ in range(T):
        W = input().strip()
        K = int(input().strip())
        positions = [[] for _ in range(26)]  # 각 알파벳 별 인덱스 기록용

        for idx, ch in enumerate(W):
            positions[ord(ch) - ord("a")].append(idx)

        shortest = float("inf")
        longest = -1

        for pos_list in positions:
            if len(pos_list) < K:
                continue

            for start in range(len(pos_list) - K + 1):
                end = start + K - 1  # start부터 K개 뒤쪽 인덱스까지 같은 문자
                length = pos_list[end] - pos_list[start] + 1  # 현재 윈도우 길이
                if length < shortest:
                    shortest = length  # 가장 짧은 정확히 K개 포함 구간 갱신
                if length > longest:
                    longest = length  # 가장 긴 정확히 K개 포함 구간 갱신

        print(-1 if longest == -1 else f"{shortest} {longest}")


if __name__ == "__main__":
    main()
