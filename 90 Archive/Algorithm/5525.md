import sys


def main() -> None:
    input = sys.stdin.readline
    N = int(input().strip())
    _ = int(input().strip())
    S = input().strip()

    answer = 0
    count = 0  # 연속으로 등장한 "IOI" 조각 수
    i = 0
    limit = len(S)

    # "IOI"를 발견할 때마다 count를 조정하고 겹칠 수 있도록 한 칸 덜 빼줌
    while i < limit - 1:
        if i + 2 < limit and S[i] == "I" and S[i + 1] == "O" and S[i + 2] == "I":
            count += 1
            if count == N:
                answer += 1
                count -= 1
            i += 2  # 다음 확인 위치는 직전 I 바로 뒤
        else:
            i += 1
            count = 0  # 패턴이 끊기면 다시 0으로 초기화

    print(answer)


if __name__ == "__main__":
    main()
